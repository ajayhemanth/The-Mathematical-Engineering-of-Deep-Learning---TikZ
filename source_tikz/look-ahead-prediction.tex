\newcommand\rnnunit[6]
{
    \foreach \i in {1,...,#6}
    {
        \ifthenelse{\i=3 \OR \i=6}
        {
            \node[thick] (unit#4\i) at (#2+\i*3,#3) {\Huge {...}};
        }
        {
            \node[draw, fill=#5, rounded corners, minimum width=2cm, minimum height=1cm] (unit#4\i) at (#2+\i*3,#3) {unit};
        }
    }
    \foreach \i in {2,...,#6}
    {
        \pgfmathparse{\i-1}\edef\ni{\pgfmathresult};
        \draw[-Triangle] (unit#4\ni) -- (unit#4\i);
    }
    
}

\newcommand\connect[3]
{
    \foreach \i in {#1,...,#2}
    {
        \pgfmathparse{\i+1}\edef\ni{\pgfmathresult};
        \draw[-Triangle,thick, bend left, looseness=2.5] (#3\i) to[out=90, in=-90] (#3\ni);
    }
}
